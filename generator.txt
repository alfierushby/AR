CREATE TABLE user (
    userID INT GENERATED ALWAYS AS IDENTITY (START WITH 1 INCREMENT BY 1) PRIMARY KEY,
    userName varchar(255) NOT NULL,
    password varchar(255) NOT NULL
);
Code for Tags Table
CREATE TABLE tags (
    userID INT,
    tagID INT GENERATED ALWAYS AS IDENTITY (START WITH 1 INCREMENT BY 1),
    tagName VARCHAR(255) NOT NULL,
    coords REAL NOT NULL,
    placed BOOLEAN NOT NULL,
    FOREIGN KEY (userID) REFERENCES user(userID) ON UPDATE RESTRICT ON DELETE RESTRICT
);
